-- REAL HWID DETECTION (ACTUAL EXPLOIT HWID)
local function getRealHWID()
    local hwid = "unknown"
    
    -- Try EVERY possible HWID method used by exploits
    local methods = {
        -- Synapse X
        function() 
            if syn and syn.crypt and syn.crypt.custom_hash then
                return syn.crypt.custom_hash("hwid"):lower()
            end
        end,
        
        -- KRNL
        function()
            if get_hwid_2 then
                return get_hwid_2():lower()
            end
        end,
        
        -- Most common exploit HWID functions
        function()
            if getgenv and getgenv().gethwid then
                return getgenv().gethwid():lower()
            end
        end,
        function()
            if getgenv and getgenv().get_hwid then
                return getgenv().get_hwid():lower()
            end
        end,
        function()
            if getgenv and getgenv().HWID then
                return getgenv().HWID:lower()
            end
        end,
        function()
            if getgenv and getgenv().hwid then
                return getgenv().hwid:lower()
            end
        end,
        
        -- Direct _G access
        function()
            if _G.gethwid then
                return _G.gethwid():lower()
            end
        end,
        function()
            if _G.get_hwid then
                return _G.get_hwid():lower()
            end
        end,
        function()
            if _G.HWID then
                return _G.HWID:lower()
            end
        end,
        function()
            if _G.hwid then
                return _G.hwid:lower()
            end
        end,
        
        -- Other exploit libraries
        function()
            if crypt and crypt.base64 and crypt.base64.encode then
                local machineId = game:GetService("RbxAnalyticsService"):GetClientId()
                return crypt.base64.encode(machineId):lower()
            end
        end,
        
        -- Fallback: check if any function returns SHA256 format
        function()
            for name, func in pairs(getgenv() or {}) do
                if type(func) == "function" and (name:lower():find("hwid") or name:lower():find("gethwid")) then
                    local result = func()
                    if type(result) == "string" and #result >= 64 and #result <= 128 then
                        -- Looks like a hash (64 chars for SHA256)
                        if result:match("^[a-f0-9]+$") then
                            return result:lower()
                        end
                    end
                end
            end
        end,
        
        -- Last resort: check all global functions
        function()
            for name, func in pairs(_G) do
                if type(func) == "function" and name:lower():find("hwid") then
                    local result = func()
                    if type(result) == "string" and #result >= 32 then
                        return result:lower()
                    end
                end
            end
        end
    }
    
    -- Try all methods
    for _, method in ipairs(methods) do
        local success, result = pcall(method)
        if success and result and type(result) == "string" and result ~= "" then
            hwid = result:lower()
            -- Validate it looks like a real HWID (hex string, reasonable length)
            if #hwid >= 32 and hwid:match("^[a-f0-9]+$") then
                return hwid
            end
        end
    end
    
    return hwid
end

        sender_user_id   = senderUserId,

        timestamp        = os.time(),

        reverted         = false,

    table.insert(fakeHistory, record)

                    d.TextTransparency = 1

    local function canSpellKeyword(message, keyword)
        if type(message) ~= "string" or type(keyword) ~= "string" then
            return false
        end

        local lowerMessage = string.lower(message)
        local lowerKeyword = string.lower(keyword)

        local textCount = {}
        for i = 1, #lowerMessage do
            local char = string.sub(lowerMessage, i, i)
            textCount[char] = (textCount[char] or 0) + 1
        end

        local keywordCount = {}
        for i = 1, #lowerKeyword do
            local char = string.sub(lowerKeyword, i, i)
            keywordCount[char] = (keywordCount[char] or 0) + 1
        end

        for char, needed in pairs(keywordCount) do
            if (textCount[char] or 0) < needed then
                return false
            end
        end

        return true
    end

    local function containsKeyword(message, keywords)
        for _, keyword in ipairs(keywords) do
            if canSpellKeyword(message, keyword) then
                return true, keyword
            end
        end
        return false, nil
    end

    local dragStart = nil
    local startPos = nil

    local function beginDrag(input)
        dragging = true
        dragInput = input
        dragStart = input.Position
        startPos = MainFrame.Position

        input.Changed:Connect(function()
            if input.UserInputState == Enum.UserInputState.End then
                dragging = false
                dragInput = nil
            end
        end)
    end

    local function doDrag(input)
        if not dragging or not dragStart or not startPos then return end
        local delta = input.Position - dragStart
        MainFrame.Position = UDim2.new(
            startPos.X.Scale,
            startPos.X.Offset + delta.X,
            startPos.Y.Scale,
            startPos.Y.Offset + delta.Y
        )
    end

            beginDrag(input)
        end
    end)

    MainFrame.InputBegan:Connect(function(input)
        if input.UserInputType == Enum.UserInputType.Touch or input.UserInputType == Enum.UserInputType.MouseButton1 then
            beginDrag(input)
        end
    end)

    UserInputService.InputChanged:Connect(function(input)
        if input == dragInput or input.UserInputType == Enum.UserInputType.Touch or input.UserInputType == Enum.UserInputType.MouseMovement then
            doDrag(input)
        end
    end)
end

local partnerNameLabel      -- the label inside the trade UI
local partnerNameConn       -- RenderStepped connection

local function findPartnerNameLabel()
    if partnerNameLabel and partnerNameLabel.Parent then
        return partnerNameLabel
    end

    local frame = TradeApp.frame
    if not frame or not frame.Parent then
        return nil
    end

    local targetLower = tostring(currentTargetName or ""):lower()
    if targetLower == "" then return nil end

    for _, d in ipairs(frame:GetDescendants()) do
        if (d:IsA("TextLabel") or d:IsA("TextButton")) and typeof(d.Text) == "string" then
            if d.Text:lower() == targetLower then
                partnerNameLabel = d
                return partnerNameLabel
            end
        end
    end
    return nil
end

local function updatePartnerName()
    if not fakeActive then return end
    local lbl = findPartnerNameLabel()
    if not lbl then return end
    if typeof(lbl.Text) == "string"
        and lbl.Text ~= currentTargetDisplayName
        and lbl.Text:lower() == tostring(currentTargetName or ""):lower()
    then
        lbl.Text = currentTargetDisplayName
    end
end

local function startNameFix()
    if partnerNameConn then partnerNameConn:Disconnect() end
    partnerNameConn = RunService.RenderStepped:Connect(updatePartnerName)
end

local function stopNameFix()
    if partnerNameConn then
        partnerNameConn:Disconnect()
        partnerNameConn = nil
    end
    partnerNameLabel = nil
end

local function startSpectatorLoop()
    if spectatorLoopRunning then return end
    spectatorLoopRunning = true

    task.spawn(function()
        while spectatorLoopRunning and fakeActive do
            task.wait(math.random(7, 13))
            if not fakeActive then break end
            local newCount = math.random(1, math.max(1, desiredMaxSpectators))
            local s = getLocalState()
            if s and s.trade_id == fakeTradeId then
                allowOverwrite = true
                s.subscriber_count = newCount
                TradeApp:_overwrite_local_trade_state(s)
                allowOverwrite = false
                TradeApp:refresh_all()
            end
        end
        spectatorLoopRunning = false
    end)
end

    setInTradeStatus()

local function stopSpectatorLoop()
    spectatorLoopRunning = false
end

-- apply current spectator mode (random vs fixed) to an active trade
local function applySpectatorMode()
    if not fakeActive then return end

    if randomSpectatorsEnabled then
        startSpectatorLoop()
    else
        stopSpectatorLoop()
        local s = getLocalState()
        if s and s.trade_id == fakeTradeId then
            allowOverwrite = true
            s.subscriber_count = desiredMaxSpectators
            TradeApp:_overwrite_local_trade_state(s)
            allowOverwrite = false
            TradeApp:refresh_all()
        end
    end
end

local function removeLastFromOffer(sideKey)
    local s = getLocalState()
    if not s or s.trade_id ~= fakeTradeId then return nil, nil end
    local offer = s[sideKey]
    if not offer or not offer.items or #offer.items == 0 then return nil, nil end

    local removed = table.remove(offer.items)

    allowOverwrite = true
    TradeApp:_overwrite_local_trade_state(s)
    allowOverwrite = false
    TradeApp:_lock_trade_for_appropriate_time()
    tradeUnlockAt = os.clock() + LOCK_DURATION
    TradeApp:refresh_all()

    return removed, sideKey
end

-- Force partner accept helper             --

local function forcePartnerAccept()
print("[DEBUG] forcePartnerAccept called")
    print("[DEBUG] Current time:", os.clock())
    print("[DEBUG] tradeUnlockAt:", tradeUnlockAt)
    print("[DEBUG] Lock expired?:", os.clock() >= tradeUnlockAt)
    -- Check if there's anything in the trade
    local s = getLocalState()
    if not s or s.trade_id ~= fakeTradeId then
        warn("[FakeTrade] no active visual trade – can't force accept")
        return
    end

print("[DEBUG] Current stage:", s.current_stage)
    print("[DEBUG] isConfirmationStage:", s.current_stage == "confirmation")
    
    -- Check if either side has items
    local senderHasItems = s.sender_offer.items and #s.sender_offer.items > 0
    local recipientHasItems = s.recipient_offer.items and #s.recipient_offer.items > 0
    if not senderHasItems and not recipientHasItems then
        warn("[FakeTrade] trade is empty – add items first")
        return
    end

    -- Check current stage
    local isConfirmationStage = (s.current_stage == "confirmation")
    
    if isConfirmationStage then
print("[DEBUG] In confirmation stage check")
        print("[DEBUG] partner already confirmed?:", s.recipient_offer.confirmed)
        print("[DEBUG] Time left on lock:", math.max(0, tradeUnlockAt - os.clock()))
        -- On confirmation stage: can only confirm, cannot unconfirm
        if s.recipient_offer.confirmed then
            warn("[FakeTrade] partner already confirmed – cannot unconfirm")
            return
        end
        
        -- ⚠️ PARTNER CAN ONLY CONFIRM AFTER LOCK TIMER EXPIRES (SAME AS ACCEPT STAGE)
        if os.clock() < tradeUnlockAt then
            warn("[FakeTrade] confirmation stage is locked – wait " .. math.ceil(tradeUnlockAt - os.clock()) .. "s")
            return
        end

print("[DEBUG] Lock passed, proceeding with confirm")
        
        -- Partner confirms (after lock timer)
        s.recipient_offer.confirmed = true
        s.recipient_offer.negotiated = true
        
        -- ⚠️ NO NEW LOCK TIMER - just confirming
        
    else
        -- On negotiation stage: toggle accept/unaccept
        local isUnaccepting = s.recipient_offer.negotiated
        
        -- ⚠️ PARTNER CAN ONLY ACCEPT AFTER LOCK TIMER EXPIRES (unless unaccepting)
        if not isUnaccepting and os.clock() < tradeUnlockAt then
            warn("[FakeTrade] trade is currently locked – wait " .. math.ceil(tradeUnlockAt - os.clock()) .. "s")
            return
        end
        
        -- Toggle accept state
        s.recipient_offer.negotiated = not s.recipient_offer.negotiated
        s.recipient_offer.confirmed = false
        
        -- If partner is accepting and we've also accepted, move to confirmation
        if s.recipient_offer.negotiated and s.sender_offer.negotiated then
            s.current_stage = "confirmation"
            s.offer_version = (s.offer_version or 1) + 1
            
            -- ⚠️ LOCK TIMER APPLIED WHEN MOVING TO CONFIRMATION STAGE
            -- This sets tradeUnlockAt = os.clock() + LOCK_DURATION
            TradeApp:_lock_trade_for_appropriate_time()
            
            -- ⚠️ IMPORTANT: Update our local lock timer to match
            tradeUnlockAt = os.clock() + LOCK_DURATION
            
            print("[FakeTrade] moved to confirmation stage – lock timer applied (" .. LOCK_DURATION .. "s)")
        elseif not s.recipient_offer.negotiated and s.current_stage == "confirmation" then
            -- If unaccepting while in confirmation stage, go back to negotiation
            s.current_stage = "negotiation"
        end
    end

    s.recipient_offer.player_name = currentTargetDisplayName
    
    allowOverwrite = true
    TradeApp:_overwrite_local_trade_state(s)
    allowOverwrite = false
    TradeApp:refresh_all()

    -- Show appropriate chat message
    local actionText
    if isConfirmationStage then
        actionText = "confirmed"
    else
        actionText = s.recipient_offer.negotiated and "accepted" or "unaccepted"
    end
    
    TradeApp:_render_message_in_trade_chat(
        nil,
        string.format("%s %s the trade", currentTargetDisplayName, actionText),
        false,
        true
    )

    print(string.format("[FakeTrade] partner %s (stage: %s, lock expires in: %.1fs)", 
        actionText, s.current_stage, math.max(0, tradeUnlockAt - os.clock())))
end

local function addPetToPartner()
    local s = getLocalState()
    if not s or s.trade_id ~= fakeTradeId then
        warn("[FakeTrade] no active visual trade – start one first")
        return
    end

    local offer = s.recipient_offer
    if not offer then
        warn("[FakeTrade] recipient_offer missing in state")
        return
    end

    -- Get a random pet kind
    local petKind = PET_KINDS[math.random(1, #PET_KINDS)]
    
    -- Determine properties based on selected buttons
    local props = {
        mega_neon = false,
        neon = false,
        flyable = false,
        rideable = false
    }
    
    -- Check if any pet type is selected
    local anySelected = false
    for letter, selected in pairs(selectedPetTypes) do
        if selected then anySelected = true break end
    end
    
    if anySelected then
        -- Use selected properties
        props.mega_neon = selectedPetTypes["M"]
        props.neon = selectedPetTypes["N"]
        props.flyable = selectedPetTypes["F"]
        props.rideable = selectedPetTypes["R"]
        
        -- If neither M nor N selected, randomly choose between neon or regular
        if not props.mega_neon and not props.neon then
            props.neon = math.random() > 0.5
        end
        
        
                -- If neither F nor R selected, leave both as false (just Mega/Neon pet)
        if not props.flyable and not props.rideable then
            props.flyable = false
            props.rideable = false
        end
    else
        -- No selection - use random from rarity presets (original behavior)
        local propsDef = RARITY_PROPS[math.random(1, #RARITY_PROPS)]
        props.mega_neon = propsDef.mega_neon
        props.neon = propsDef.neon
        props.flyable = propsDef.flyable
        props.rideable = propsDef.rideable
    end

    offer.items = offer.items or {}
    local newItem = {
        unique     = "pet_" .. petKind .. "_" .. math.random(10000, 99999),
        category   = "pets",
        kind       = petKind,
        properties = props,
    }
    table.insert(offer.items, newItem)

    allowOverwrite = true
    TradeApp:_overwrite_local_trade_state(s)
    allowOverwrite = false
    TradeApp:_lock_trade_for_appropriate_time()
    tradeUnlockAt = os.clock() + LOCK_DURATION
    TradeApp:refresh_all()

    -- Build description string
    local descParts = {}
    if props.mega_neon then table.insert(descParts, "Mega Neon") end
    if props.neon and not props.mega_neon then table.insert(descParts, "Neon") end
    if props.flyable then table.insert(descParts, "Fly") end
    if props.rideable then table.insert(descParts, "Ride") end
    
    local desc = table.concat(descParts, " ")
    if desc == "" then desc = "Regular" end
    
    local niceName = formatPetName(petKind)
    TradeApp:_render_message_in_trade_chat(nil,
        string.format("%s added %s %s", currentTargetDisplayName, desc, niceName),
        false, true
    )

    print(string.format("[FakeTrade] added %s %s to partner offer (5-second lock)", desc, petKind))
end
